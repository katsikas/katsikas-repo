 70function [c,ia,ib] = do_vectors(opcode,a,b)
 71% Union of the vectors A and B.
 72coder.inline('always');
 73eml_allow_enum_inputs;
 74ZERO = zeros(eml_index_class);
 75ONE = ones(eml_index_class);
 76na = cast(eml_numel(a),eml_index_class);
 77nb = cast(eml_numel(b),eml_index_class);
 78% We intentionally leave in the much of the logic required to handle [],
 79% column vectors, and vector inputs with different orientations. In future
 80% we may have consistent behavior to match and also be able to support it
 81% efficiently, at which time the extra code will provide a convenient
 82% starting point. Currently, the compiler will constant-fold isCol and
 83% elide the extra code.
 84isNullA = false; % isequal(a,[]);
 85isNullB = false; % isequal(b,[]);
 86isVectorA = eml_is_const(isvector(a)) && isvector(a);
 87isVectorB = eml_is_const(isvector(b)) && isvector(b);
 88isScalarA = eml_is_const(isscalar(a)) && isscalar(a);
 89isScalarB = eml_is_const(isscalar(b)) && isscalar(b);
 90isColA = eml_is_const(size(a,2)) && size(a,2) == 1 && ~isScalarA;
 91isColB = eml_is_const(size(b,2)) && size(b,2) == 1 && ~isScalarB;
 92eml_invariant(isVectorA && ~isColA && isVectorB && ~isColB, ...
 93    eml_message('Coder:toolbox:eml_setop_9'), ...
 94    'IfNotConst','Fail');
 95if opcode == INTERSECT
 96    CZERO = eml_scalar_eg([eml_scalar_eg(a),eml_scalar_eg(b)]);
 97    ncmax = min(na,nb);
 98    niamax = ncmax;
 99    nibmax = ncmax;
100elseif opcode == SETDIFF
101    CZERO = eml_scalar_eg(a);
102    ncmax = na;
103    niamax = na;
104    nibmax = ZERO;
105else % opcode == SETXOR || opcode == UNION
106    CZERO = eml_scalar_eg([eml_scalar_eg(a),eml_scalar_eg(b)]);
107    ncmax = eml_index_plus(na,nb);
108    niamax = na;
109    nibmax = nb;
110end
111if opcode == SETDIFF
112    isCol = isColA || ((isScalarA || isNullA) && isColB);
113else
114    isCol = ...
115        (isColA && isColB) || ...
116        (isColA && (isScalarB || isNullB)) || ...
117        (isColB && (isScalarA || isNullA));
118end
119if eml_const(isCol)
120    coder.varsize('c','ia','ib',[],[1,0]);
121    c = coder.nullcopy(eml_expand(CZERO,[ncmax,1]));
122    ia = coder.nullcopy(zeros(niamax,1));
123else
124    coder.varsize('c','ia','ib',[],[0,1]);
125    c = coder.nullcopy(eml_expand(CZERO,[1,ncmax]));
126    ia = coder.nullcopy(zeros(1,niamax));
127end
128nc = ZERO;
129nia = ZERO;
130if isCol
131    ib = coder.nullcopy(zeros(nibmax,1));
132else
133    ib = coder.nullcopy(zeros(1,nibmax));
134end
135nib = ZERO;
136if ~issorted(a)
137    eml_error('Coder:toolbox:eml_setop_unsortedA');
138end
139if ~issorted(b)
140    eml_error('Coder:toolbox:eml_setop_unsortedB');
141end
142i1 = ONE;
143i2 = ONE;
144while i1 <= na && i2 <= nb
145    ak = a((i1));
146    % Skip forward to the last instance of this value of ak.
147    while i1 < na
148        if eml_safe_eq(a((eml_index_plus(i1,1))),ak)
149            i1 = eml_index_plus(i1,1);
150        else
151            break
152        end
153    end
154    bk = b((i2));
155    % Skip forward to the last instance of this value of bk.
156    while i2 < nb
157        if eml_safe_eq(b((eml_index_plus(i2,1))),bk)
158            i2 = eml_index_plus(i2,1);
159        else
160            break
161        end
162    end
163    if eml_safe_eq(ak,bk)
164        if opcode == INTERSECT
165            nc = eml_index_plus(nc,1);
166            c(nc) = ak;
167            ia(nc) = (i1);
168            ib(nc) = (i2);
169        elseif opcode == UNION
170            nc = eml_index_plus(nc,1);
171            c(nc) = ak;
172            % MATLAB UNION does not note when elements of A are also in B.
173            % nia = eml_index_plus(nia,1);
174            % ia(nia) = (i1);
175            nib = eml_index_plus(nib,1);
176            ib(nib) = (i2);
177        end
178        i1 = eml_index_plus(i1,1);
179        i2 = eml_index_plus(i2,1);
180    elseif eml_sort_lt(ak,bk) % ak < bk
181        if opcode == SETDIFF || opcode == SETXOR || opcode == UNION
182            nc = eml_index_plus(nc,1);
183            c(nc) = ak;
184            nia = eml_index_plus(nia,1);
185            ia(nia) = (i1);
186        end
187        i1 = eml_index_plus(i1,1);
188    else
189        if opcode == SETXOR
190            nc = eml_index_plus(nc,1);
191            c(nc) = bk;
192            nib = eml_index_plus(nib,1);
193            ib(nib) = (i2);
194        elseif opcode == UNION
195            nc = eml_index_plus(nc,1);
196            c(nc) = bk;
197            nib = eml_index_plus(nib,1);
198            ib(nib) = (i2);
199        end
200        i2 = eml_index_plus(i2,1);
201    end
202end
203if opcode == SETDIFF || opcode == SETXOR || opcode == UNION
204    % Remaining elements of A are not present in B.
205    while i1 <= na
206        ak = a((i1));
207        while i1 < na
208            if eml_safe_eq(a((eml_index_plus(i1,1))),ak)
209                i1 = eml_index_plus(i1,1);
210            else
211                break
212            end
213        end
214        nc = eml_index_plus(nc,1);
215        c(nc) = ak;
216        nia = eml_index_plus(nia,1);
217        ia(nia) = (i1);
218        i1 = eml_index_plus(i1,1);
219    end
220end
221if opcode == SETXOR || opcode == UNION
222    % Remaining elements of B are not present in A.
223    while i2 <= nb
224        bk = b((i2));
225        while i2 < nb
226            if eml_safe_eq(b((eml_index_plus(i2,1))),bk)
227                i2 = eml_index_plus(i2,1);
228            else
229                break
230            end
231        end
232        nc = eml_index_plus(nc,1);
233        c(nc) = bk;
234        nib = eml_index_plus(nib,1);
235        ib(nib) = (i2);
236        i2 = eml_index_plus(i2,1);
237    end
238end
239if opcode == INTERSECT
240    nia = nc;
241    nib = nc;
242end
243% Trim the output arrays.
244if ncmax > 0
245    assert(nc <= ncmax); %<HINT>
246    if isCol
247        c = c(1:nc,1);
248    else
249        c = c(1,1:nc);
250    end
251end
252if niamax > 0
253    assert(nia <= niamax); %<HINT>
254    if isCol
255        ia = ia(1:nia,1);
256    else
257        ia = ia(1,1:nia);
258    end
259end
260if opcode == INTERSECT || opcode == SETXOR || opcode == UNION
261    if nibmax > 0
262        assert(nib <= nibmax); %<HINT>
263        if isCol
264            ib = ib(1:nib,1);
265        else
266            ib = ib(1,1:nib);
267        end
268    end
269end
